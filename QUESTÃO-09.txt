#include <stdio.h>

struct Veiculo {
    char placa[50];
    char marca[50];
    char modelo[50];
    int ano;
    float km;
};

void lerVeiculo(struct Veiculo v[], int n) {
    for (int i = 0; i < n; i++) {

        printf("\n---- Veiculo %d ----\n", i + 1);

        printf("Placa do veiculo: ");
        scanf(" %49[^\n]", v[i].placa);

        printf("Marca do veiculo: ");
        scanf(" %49[^\n]", v[i].marca);

        printf("Modelo do veiculo: ");
        scanf(" %49[^\n]", v[i].modelo);

        printf("Ano do veiculo: ");
        scanf("%d", &v[i].ano);

        printf("Km do veiculo: ");
        scanf("%f", &v[i].km);
    }
}

void exibirVeiculo(struct Veiculo v) {
    printf("\nPlaca: %s", v.placa);
    printf("\nMarca: %s", v.marca);
    printf("\nModelo: %s", v.modelo);
    printf("\nAno: %d", v.ano);
    printf("\nKm: %.2f\n", v.km);
}

void insertionSort(struct Veiculo v[], int n) {
    struct Veiculo chave;
    int j;

    for (int i = 1; i < n; i++) {

        chave = v[i];
        j = i - 1;

        while (j >= 0 &&
              (v[j].ano < chave.ano ||
              (v[j].ano == chave.ano && v[j].km > chave.km))) {

            v[j+1] = v[j];
            j--;
        }

        v[j+1] = chave;
    }
}

void exibirLista(struct Veiculo v[], int n) {
    printf("\n--- Insert Sort (Veiculo) ---\n");

    for (int i = 0; i < n; i++)
        exibirVeiculo(v[i]);
}

int main() {
    int n;

    printf("Quantos veiculos deseja cadastrar? ");
    scanf("%d", &n);

    struct Veiculo veiculos[n];

    lerVeiculo(veiculos, n);

    insertionSort(veiculos, n);

    exibirLista(veiculos, n);
}
