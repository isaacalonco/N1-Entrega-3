#include <stdio.h>
#include <string.h>

struct Funcionario {
    char nome[50];
    char cargo[50];
    float salario;
};

int main() {

    int n = 5;
    struct Funcionario f[50];
    struct Funcionario temp;
    float soma = 0, media;

    //
    for(int i = 0; i < n; i++) {

        printf("\n--- Funcionario %d ---\n", i + 1);

        printf("Nome: ");
        scanf(" %[^\n]", f[i].nome);

        printf("Cargo: ");
        scanf(" %[^\n]", f[i].cargo);

        printf("Salario: ");
        scanf("%f", &f[i].salario);

        soma += f[i].salario;
    }

    // Aqui é realizado o cálculo da média salarial.
    media = soma / n;

    // Aqui é feito o bubble sort por ordem alfabética
    for(int i = 0; i < n - 1; i++) {
        for(int j = 0; j < n - i - 1; j++) {

            if(strcmp(f[j].nome, f[j+1].nome) > 0) {

                temp = f[j];
                f[j] = f[j+1];
                f[j+1] = temp;

            }
        }
    }


    printf("\nFuncionarios em ordem alfabetica:\n");

    for(int i = 0; i < n; i++) {

        printf("\nNome: %s\n", f[i].nome);
        printf("Cargo: %s\n", f[i].cargo);
        printf("Salario: %.2f\n", f[i].salario);

        if(f[i].salario >= media)
            printf("Acima da media salarial\n");
        else
            printf("Abaixo da media salarial\n");
    }

    printf("\nMedia salarial: %.2f\n", media);

    return 0;
}
