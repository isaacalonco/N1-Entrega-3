#include <stdio.h>

struct Pedido { int numeroPedido; char nomeCliente[80]; int data; float valorTotal; };

void lerPedidos(struct Pedido pedido[], int n) {
    int i;
    for (i = 0; i < n; i++) {
        printf("\n---- PEDIDO %d ----\n", i + 1);
        printf("Numero do pedido: ");
        scanf("%d", &pedido[i].numeroPedido);
        printf("Nome do cliente: ");
        scanf(" %[^\n]", pedido[i].nomeCliente);
        printf("Data (AAAAMMDD): ");
        scanf("%d", &pedido[i].data);
        printf("Valor total: ");
        scanf("%f", &pedido[i].valorTotal);
    }
}

void insertionSort(struct Pedido pedido[], int n) {
    int i, j;
    struct Pedido aux;
    for (i = 1; i < n; i++) {
        aux = pedido[i];
        j = i - 1;
        while (j >= 0 && pedido[j].data > aux.data) {
            pedido[j + 1] = pedido[j];
            j--;
        }
        pedido[j + 1] = aux;
        printf("\nPassada %d realizada\n", i);
    }
}

void mostrar(struct Pedido pedido[], int n) {
    int i;
    float totalAcumulado = 0;
    printf("\n---- PEDIDOS ORDENADOS POR DATA ----\n");
    for (i = 0; i < n; i++) {
        printf("\nNumero: %d", pedido[i].numeroPedido);
        printf("\nCliente: %s", pedido[i].nomeCliente);
        printf("\nData: %d", pedido[i].data);
        printf("\nValor: R$ %.2f\n", pedido[i].valorTotal);
        totalAcumulado += pedido[i].valorTotal;
    }
    printf("\n---- TOTAL ACUMULADO: R$ %.2f ----\n", totalAcumulado);
}

int main() {
    int n;

    printf("Quantos pedidos? ");
    scanf("%d", &n);

    struct Pedido pedido[n];
    lerPedidos(pedido, n);
    insertionSort(pedido, n);
    mostrar(pedido, n);
}
