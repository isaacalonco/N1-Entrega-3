#include <stdio.h>
#include <stdlib.h>


struct Livro{
	char titulo[20];
	char autor[20];
	int anoPublicacao;
	int numPaginas;
};
void Order(struct Livro livros[],int *n,int *swap){
	struct Livro aux;
	int i, j;

	for (i = 0; i < *n - 1; i++) {
		for (j = 0; j < *n - i - 1; j++) {

			if (livros[j].anoPublicacao > livros[j+1].anoPublicacao) {
				aux    = livros[j];
				livros[j]   = livros[j+1];
				livros[j+1] = aux;
				(*swap)++;
			}
		}
	}
}
int main() {
	int n;
	printf("Digite a quantidade de livros para registrar: \n");
	scanf("%d",&n);
	struct Livro livros[n];

	int i,swap=0;
	for(i=0;i<n;i++){
		printf("digite o titulo do livro %d: \n",i+1);
		scanf(" %19[^\n]",livros[i].titulo);

		printf("digite o autor do livro %d : \n",i+1);
		scanf(" %19[^\n]",livros[i].autor);

		printf("digite o ano da publicacao do livro %d: \n",i+1);
		scanf("%d",&livros[i].anoPublicacao);

		printf("digite o numero de paginas do livro %d: \n",i+1);
		scanf("%d",&livros[i].numPaginas);
	}
	Order(livros,&n,&swap);

	for(i=0;i<n;i++){
			printf("\n-----Livro %d-----\n",i+1);
			printf("Titulo: %s \n",livros[i].titulo);
			printf("Autor: %s \n",livros[i].autor);
			printf("Ano publicacao: %d \n",livros[i].anoPublicacao);
			printf("Numero de paginas: %d \n",livros[i].numPaginas);
	}
	printf("Quantidade de trocas realizadas: %d",swap);
}
